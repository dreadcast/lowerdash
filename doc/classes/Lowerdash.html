eachAsync

(
  • obj
  • iterator
  • [cb]
  • [bind]
)
Object

Defined in src/lowerdash.js:191

Wait next cursor to be called before next iteration

Parameters:

  • obj Object

    Object or array.

  • iterator Function

    Iterator, invoked on each entry. Passed arguments are item, key (if passed obj is iterable), index, cursor and passed arr.

  • [cb] Function optional

    Callback invoked after last iteration.

  • [bind] Object optional

    Object bound to iterator, default to passed arr.

Returns:

Object:

arr Passed object

Example:

_.eachAsync([1, 2, 3, 4], function(item, index, cursor, ar){
    new Request({
        url: '/rest/' + index,
        onSuccess: function(){
            console.info('item #' + index + ' posted');
            cursor();
        }
    }).post();
}, function(){ console.info('complete'); })

// Works with an array of functions...
_.eachAsync([function(cursor){
    new Request({
        url: '/some/url/123',
        onSuccess: cursor
    }).get();
}, function(cursor){
    new Request({
        url: '/some/url/456',
        onSuccess: cursor
    }).get();
}], function(item, index, cursor, ar){
    item(cursor);
});
